1. Counter Component

The simplest component — state, events, render. Notice how Nectar eliminates the hook ceremony and runtime overhead.

Counter.jsx — React 22 lines 
import { useState } from 'react';

export default function Counter({ initial = 0 }) {
  const [count, setCount] = useState(initial);

  return (
    <div>
      <h2>Counter</h2>
      <span>{count}</span>
      <button onClick={() => setCount(c => c + 1)}>
        +1
      </button>
      <button onClick={() => setCount(c => c - 1)}>
        -1
      </button>
    </div>
  );
}

// Every click: React diffs the entire virtual DOM tree
// to find that one <span> changed. O(n) tree walk.
counter.nectar — Arc 17 lines 
component Counter(initial: i32) {
    let mut count: i32 = initial;

    fn increment(&mut self) {
        self.count = self.count + 1;
    }

    fn decrement(&mut self) {
        self.count = self.count - 1;
    }

    render {
        <div>
            <h2>"Counter"</h2>
            <span>{self.count}</span>
            <button on:click={self.increment}>"+1"</button>
            <button on:click={self.decrement}>"-1"</button>
        </div>
    }
}

// Click: signal updates ONE DOM node directly. O(1).
// No VDOM diff. No GC. Runs as native WASM.
▶ Why Nectar wins here: No useState / setCount ceremony. State is just let mut. Methods are real functions, not closures recreated every render. The signal system patches exactly the one <span> that displays count — no tree diffing.
Nectar advantage

2. Todo App — Structs, Enums, Filtering

A real-world CRUD component. React needs useCallback, useMemo, and careful reference management. Nectar just uses ownership.

TodoApp.jsx — React 62 lines 
import { useState, useMemo, useCallback } from 'react';

const FILTERS = { All: 'all', Active: 'active', Done: 'done' };

export default function TodoApp() {
  const [todos, setTodos]     = useState([]);
  const [nextId, setNextId]   = useState(0);
  const [filter, setFilter]   = useState(FILTERS.All);
  const [input, setInput]     = useState('');

  // Must memoize or every child re-renders
  const addTodo = useCallback(() => {
    setTodos(prev => [...prev, {
      id: nextId,
      text: input,
      done: false
    }]);
    setNextId(n => n + 1);
    setInput('');
  }, [input, nextId]);

  const toggle = useCallback((id) => {
    setTodos(prev =>
      prev.map(t =>
        t.id === id ? { ...t, done: !t.done } : t
      )
    );
  }, []);

  // Must memoize filtered list too
  const visible = useMemo(() => {
    if (filter === FILTERS.Active) return todos.filter(t => !t.done);
    if (filter === FILTERS.Done)   return todos.filter(t => t.done);
    return todos;
  }, [todos, filter]);

  return (
    <div>
      <h1>Todos</h1>
      <input
        value={input}
        onChange={e => setInput(e.target.value)}
        placeholder="What needs to be done?"
      />
      <button onClick={addTodo}>Add</button>
      <div>
        {Object.values(FILTERS).map(f => (
          <button key={f} onClick={() => setFilter(f)}>
            {f}
          </button>
        ))}
      </div>
      <ul>
        {visible.map(todo => (
          <li key={todo.id} onClick={() => toggle(todo.id)}>
            {todo.done ? '✓ ' : ''}{todo.text}
          </li>
        ))}
      </ul>
    </div>
  );
}
todo.nectar — Arc 46 lines 
struct Todo {
    id: u32,
    text: String,
    done: bool,
}

enum Filter { All, Active, Completed }

component TodoApp() {
    let mut todos: [Todo] = [];
    let mut next_id: u32 = 0;
    let mut filter: Filter = Filter::All;

    fn add_todo(&mut self, text: String) {
        let todo = Todo {
            id: self.next_id,
            text: text,
            done: false,
        };
        self.next_id = self.next_id + 1;
        // Ownership: todo is moved into the vec
        self.todos.push(todo);
    }

    fn toggle(&mut self, id: u32) {
        for todo in &mut self.todos {
            if todo.id == id {
                todo.done = !todo.done;
            }
        }
    }

    fn visible_todos(&self) -> [&Todo] {
        match self.filter {
            Filter::All       => &self.todos,
            Filter::Active    => self.todos.iter()
                .filter(fn(t: &Todo) { !t.done }),
            Filter::Completed => self.todos.iter()
                .filter(fn(t: &Todo) { t.done }),
        }
    }

    render {
        <div>
            <h1>"Nectar Todo"</h1>
            <input placeholder="What needs to be done?" />
            <button on:click={self.add_todo}>"Add"</button>
            <ul>{self.visible_todos()}</ul>
        </div>
    }
}
▶ Key differences: React needs useCallback + dependency arrays to avoid stale closures and unnecessary re-renders. Nectar's methods are real methods with &mut self — the borrow checker guarantees correctness at compile time. match on an enum is exhaustive — add a variant and the compiler catches every missed case. React's string filter has no such safety net.
Nectar advantage

auth.nectar — Arc 32 lines, one file 
enum AuthStatus {
    LoggedOut,
    Loading,
    LoggedIn(User),
    Error(String),
}

store AuthStore {
    signal status: AuthStatus = AuthStatus::LoggedOut;
    signal token: String = "";

    action logout(&mut self) {
        self.status = AuthStatus::LoggedOut;
        self.token = "";
    }

    async action login(&mut self, email: String, pw: String) {
        self.status = AuthStatus::Loading;
        let res = await fetch("/api/login", {
            method: "POST",
            body: format("{\"email\":\"{}\",\"pw\":\"{}\"}", email, pw),
        });
        match res.status {
            200 => {
                self.token = res.headers.get("Authorization");
                self.status = AuthStatus::LoggedIn(res.json());
            }
            _ => self.status = AuthStatus::Error("Login failed"),
        }
    }

    computed is_logged_in(&self) -> bool {
        match self.status {
            AuthStatus::LoggedIn(_) => true,
            _ => false,
        }
    }
}

// Use from any component: AuthStore::login(email, pw)
// No Provider, no useSelector, no dispatch
▶ What's eliminated: No Redux, no slices, no thunks, no Provider tree, no useSelector/useDispatch. store is a language primitive with built-in signal, action, computed, and effect. Any component calls AuthStore::login() directly — type-checked at compile time.
Nectar advantage

4. Async Data Fetching + Error Handling

React's hook rules make async code awkward. Nectar uses Rust's Result type and ? operator.

usePosts.js + PostList.jsx — React 45 lines 
function usePosts() {
  const [posts, setPosts]     = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError]     = useState(null);

  const fetchPosts = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch('/api/posts');
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      setPosts(await res.json());
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { fetchPosts(); }, [fetchPosts]);

  return { posts, loading, error, refetch: fetchPosts };
}

// 3 state variables, try/catch, useCallback, useEffect,
// dependency arrays — all for one GET request.
// Errors are untyped strings. Null checks everywhere.
posts.nectar — Arc 25 lines 
store PostService {
    signal posts: [Post] = [];
    signal loading: bool = false;
    signal error: Option<ApiError> = None;

    async action fetch_posts(&mut self) {
        self.loading = true;
        self.error = None;

        let res = await fetch("/api/posts");

        match res.status {
            200 => self.posts = res.json(),
            code => self.error = Some(ApiError {
                status: code,
                message: "Failed to fetch",
            }),
        }

        self.loading = false;
    }

    computed post_count(&self) -> u32 {
        self.posts.len()
    }
}

// Errors are typed structs, not thrown strings.
// No useEffect. No dependency arrays. No stale closures.
▶ Error handling: React: errors are thrown strings caught in try/catch — no type, no structure, easy to forget. Nectar: errors are Option<ApiError> — a typed struct. The compiler forces you to handle both cases in match. No useEffect dance, no dependency arrays, no stale closure bugs.
Nectar advantage

agent WeatherChat {
    model: "claude-sonnet-4-6",
    system: "You are a helpful weather assistant.",

    tool get_weather(city: String) -> Weather {
        let res = await fetch(
            f"/api/weather?city={city}"
        );
        res.json()
    }
}

component ChatUI() {
    let mut agent = WeatherChat::new();
    let mut input: String = "";

    fn send(&mut self) {
        let reply = await self.agent.chat(self.input);
        self.input = "";
    }

    render {
        <div>
            {for msg in self.agent.history {
                <p>{msg.content}</p>
            }}
            <input value={self.input} />
            <button on:click={self.send}>"Send"</button>
        </div>
    }
}

// The tool loop is handled by the language runtime.
// Tools are type-checked. History is managed automatically.
▶ AI as a first-class citizen: agent, tool, and prompt are language keywords, not library imports. The Nectar runtime handles the tool-use loop automatically — call agent.chat() and get back the final answer. Tool parameters are typed and validated at compile time. No JSON schema objects, no manual dispatch.

6. Routing

Both approaches are declarative. Nectar bakes it in; React needs react-router.

App.jsx — React Router 24 lines + npm install 
import { BrowserRouter, Routes, Route } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/"
          element={<Home />} />
        <Route path="/dashboard"
          element={<Dashboard />} />
        <Route path="/users/:id"
          element={<UserProfile />} />
        <Route path="*"
          element={<NotFound />} />
      </Routes>
    </BrowserRouter>
  );
}

// + npm install react-router-dom (21 KB gzip)
app.nectar — Arc 16 lines, zero deps 
router AppRouter {
    route "/"              => Home,
    route "/dashboard"     => Dashboard {
        guard: AuthStore::is_logged_in,
    },
    route "/users/:id"     => UserProfile,
    fallback               => NotFound,
}

component App() {
    render {
        <AppRouter />
    }
}

// Built-in. Route guards are type-checked.
// No npm install. 0 KB added to bundle.
▶ Similar DX, different tradeoffs: Both are declarative. React Router is mature and battle-tested. Nectar's router keyword means zero extra dependencies, and guard: is compiled — a typo in the guard function name is a compile error, not a runtime 404.
